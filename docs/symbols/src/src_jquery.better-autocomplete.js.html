 27 /**
 28  * Create or alter an autocomplete object instance that belongs to
 29  * the elements in the selection. Make sure there are only text field elements
 30  * in the selection.
 31  *
 32  * @constructor
 33  *
 34  * @name jQuery.betterAutocomplete
 35  *
 36  * @param {String} method
 37  *   Should be one of the following:
 38  *   <ul><li>
 39  *     init: Initiate Better Autocomplete instances on the text input elements
 40  *     in the current jQuery selection. They are enabled by default. The other
 41  *     parameters are then required.
 42  *   </li><li>
 43  *     enable: In this jQuery selection, reenable the Better Autocomplete
 44  *     instances.
 45  *   </li><li>
 46  *     disable: In this jQuery selection, disable the Better Autocomplete
 47  *     instances.
 48  *   </li><li>
 49  *     destroy: In this jQuery selection, destroy the Better Autocomplete
 50  *     instances. It will not be possible to reenable them after this.
 51  *   </li></ul>
 52  *
 53  * @param {String|Object} [resource]
 54  *   If String, it will become the path for a remote resource. If not, it will
 55  *   be treated like a local resource. The path should provide JSON objects
 56  *   upon HTTP requests.
 57  *
 58  * @param {Object} [options]
 59  *   An object with configurable options:
 60  *   <ul><li>
 61  *     charLimit: (default=3 for remote or 1 for local resource) The minimum
 62  *     number of chars to do an AJAX call. A typical use case for this limit is
 63  *     to reduce server load.
 64  *   </li><li>
 65  *     delay: (default=350) The time in ms between last keypress and AJAX call.
 66  *     Typically used to prevent looking up irrelevant strings while the user
 67  *     is still typing. Only relevant for remote resources.
 68  *   </li><li>
 69  *     caseSensitive: (default=false) If the search should be case sensitive.
 70  *     If false, query strings will be converted to lowercase.
 71  *   </li><li>
 72  *     cacheLimit: (default=256 for remote or 0 for local resource) The maximum
 73  *     number of result objects to store in the cache. This option reduces
 74  *     server load if the user deletes characters to check back on previous
 75  *     results. To disable caching of previous results, set this option to 0.
 76  *   </li><li>
 77  *     remoteTimeout: (default=10000) The timeout for remote (AJAX) calls.
 78  *   </li><li>
 79  *     crossOrigin: (default=false) Set to true if cross origin requests will
 80  *     be performed, i.e. that the remote URL has a different domain. This will
 81  *     force Internet Explorer to use "jsonp" instead of "json" as datatype.
 82  *   </li><li>
 83  *     selectKeys: (default=[9, 13]) The key codes for keys which will select
 84  *     the current highlighted element. The defaults are tab, enter.
 85  *   </li></ul>
 86  *
 87  * @param {Object} [callbacks]
 88  *   An object containing optional callback functions on certain events. See
 89  *   {@link callbacks} for details. These callbacks should be used when
 90  *   customization of the default behavior of Better Autocomplete is required.
 91  *
 92  * @returns {Object}
 93  *   The jQuery object with the same element selection, for chaining.
 94  */
 95 
 96 (function($) {
 97 
 98 $.fn.betterAutocomplete = function(method) {
 99 
100   /*
101    * Each method expects the "this" object to be a valid DOM text input node.
102    * The methods "enable", "disable" and "destroy" expects an instance of a
103    * BetterAutocomplete object as their first argument.
104    */
105   var methods = {
106     init: function(resource, options, callbacks) {
107       var $input = $(this),
108         bac = new BetterAutocomplete($input, resource, options, callbacks);
109       $input.data('better-autocomplete', bac);
110       bac.enable();
111     },
112     enable: function(bac) {
113       bac.enable();
114     },
115     disable: function(bac) {
116       bac.disable();
117     },
118     destroy: function(bac) {
119       bac.destroy();
120     }
121   };
122 
123   var args = Array.prototype.slice.call(arguments, 1);
124 
125   // Method calling logic
126   this.each(function() {
127     switch (method) {
128     case 'init':
129       methods[method].apply(this, args);
130       break;
131     case 'enable':
132     case 'disable':
133     case 'destroy':
134       var bac = $(this).data('better-autocomplete');
135       if (bac instanceof BetterAutocomplete) {
136         methods[method].call(this, bac);
137       }
138       break;
139     default:
140       $.error(['Method', method,
141           'does not exist in jQuery.betterAutocomplete.'].join(' '));
142     }
143   });
144 
145   // Maintain chainability
146   return this;
147 };
148 
149 /**
150  * The BetterAutocomplete constructor function. Returns a BetterAutocomplete
151  * instance object.
152  *
153  * @private @constructor
154  * @name BetterAutocomplete
155  *
156  * @param {Object} $input
157  *   A single input element wrapped in jQuery.
158  */
159 var BetterAutocomplete = function($input, resource, options, callbacks) {
160 
161   var lastRenderedQuery = '',
162     cache = {}, // Key-valued caching of search results
163     cacheOrder = [], // Array of query strings, in the order they are added
164     cacheSize = 0, // Keep count of the cache's size
165     timer, // Used for options.delay
166     activeRemoteCalls = [], // A flat array of query strings that are pending
167     disableMouseHighlight = false, // Suppress the autotriggered mouseover event
168     inputEvents = {},
169     isLocal = ($.type(resource) != 'string');
170 
171   options = $.extend({
172     charLimit: isLocal ? 1 : 3,
173     delay: 350, // milliseconds
174     caseSensitive: false,
175     cacheLimit: isLocal ? 0 : 256, // Number of result objects
176     remoteTimeout: 10000, // milliseconds
177     crossOrigin: false,
178     selectKeys: [9, 13] // [tab, enter]
179   }, options);
180 
181   callbacks = $.extend({}, defaultCallbacks, callbacks);
182 
183   var $results = $('<ul />')
184     .addClass('better-autocomplete');
185 
186   callbacks.insertSuggestionList($results, $input, options.maxHeight);
187 
188   inputEvents.focus = function() {
189     redraw(true);
190   };
191 
192   inputEvents.blur = function() {
193     redraw();
194   };
195 
196   inputEvents.keydown = function(event) {
197     var index = getHighlighted();
198     // If an arrow key is pressed and a result is highlighted
199     if ($.inArray(event.keyCode, [38, 40]) >= 0 && index >= 0) {
200       var newIndex,
201         size = $('.result', $results).length;
202       switch (event.keyCode) {
203       case 38: // Up arrow
204         newIndex = Math.max(0, index - 1);
205         break;
206       case 40: // Down arrow
207         newIndex = Math.min(size - 1, index + 1);
208         break;
209       }
210       disableMouseHighlight = true;
211       setHighlighted(newIndex, true);
212       return false;
213     }
214     // A select key has been pressed
215     else if ($.inArray(event.keyCode, options.selectKeys) >= 0 &&
216              !event.shiftKey && !event.ctrlKey && !event.altKey &&
217              !event.metaKey) {
218       select();
219       return event.keyCode == 9; // Never cancel tab
220     }
221   };
222 
223   inputEvents.keyup = inputEvents.click = reprocess;
224 
225   $('.result', $results[0]).live({
226     // When the user hovers a result with the mouse, highlight it.
227     mouseover: function() {
228       if (disableMouseHighlight) {
229         return;
230       }
231       setHighlighted($('.result', $results).index($(this)));
232     },
233     mousemove: function() {
234       // Enable mouseover again.
235       disableMouseHighlight = false;
236     },
237     mousedown: function() {
238       select();
239       return false;
240     }
241   });
242 
243   // Prevent blur when clicking on group titles, scrollbars etc.,
244   // This event is triggered after the others' because of bubbling order.
245   $results.mousedown(function() {
246     return false;
247   });
248 
249   /*
250    * PUBLIC METHODS
251    */
252 
253   /**
254    * Enable this instance.
255    */
256   this.enable = function() {
257     // Turn off the browser's autocompletion
258     $input
259       .attr('autocomplete', 'OFF')
260       .attr('aria-autocomplete', 'list');
261     $input.bind(inputEvents);
262   };
263 
264   /**
265    * Disable this instance.
266    */
267   this.disable = function() {
268     $input
269       .removeAttr('autocomplete')
270       .removeAttr('aria-autocomplete');
271     $results.hide();
272     $input.unbind(inputEvents);
273   };
274 
275   /**
276    * Disable and remove this instance. This instance should not be reused.
277    */
278   this.destroy = function() {
279     $results.remove();
280     $input.unbind(inputEvents);
281     $input.removeData('better-autocomplete');
282   };
283 
284   /*
285    * PRIVATE METHODS
286    */
287 
288   /**
289    * Add an array of results to the cache. Internal methods always reads from
290    * the cache, so this method must be invoked even when caching is not used,
291    * e.g. when using local results. This method automatically clears as much of
292    * the cache as required to fit within the cache limit.
293    *
294    * @param {String} query
295    *   The query to set the results to.
296    *
297    * @param {Array[Object]} results
298    *   The array of results for this query.
299    */
300   var cacheResults = function(query, results) {
301     cacheSize += results.length;
302     // Now reduce size until it fits
303     while (cacheSize > options.cacheLimit && cacheOrder.length) {
304       var key = cacheOrder.shift();
305       cacheSize -= cache[key].length;
306       delete cache[key];
307     }
308     cacheOrder.push(query);
309     cache[query] = results;
310   };
311 
312   /**
313    * Set highlight to a specific result item
314    *
315    * @param {Number} index
316    *   The result's index, starting at 0.
317    *
318    * @param {Boolean} [autoScroll]
319    *   (default=false) If scrolling of the results list should be automated.
320    */
321   var setHighlighted = function(index, autoScroll) {
322     // Scrolling upwards
323     var up = index == 0 || index < getHighlighted();
324     var $scrollTo = $('.result', $results)
325       .removeClass('highlight')
326       .eq(index).addClass('highlight');
327 
328     if (!autoScroll) {
329       return;
330     }
331     // Scrolling up, then make sure to show the group title
332     if ($scrollTo.prev().is('.group') && up) {
333       $scrollTo = $scrollTo.prev();
334     }
335     // Is $scrollTo partly above the visible region?
336     if ($scrollTo.position().top < 0) {
337       $results.scrollTop($scrollTo.position().top + $results.scrollTop());
338     }
339     // Or is it partly below the visible region?
340     else if (($scrollTo.position().top + $scrollTo.outerHeight()) >
341               $results.height()) {
342       $results.scrollTop($scrollTo.position().top + $results.scrollTop() +
343           $scrollTo.outerHeight() - $results.height());
344     }
345   };
346 
347   /**
348    * Retrieve the index of the currently highlighted result item
349    *
350    * @returns {Number}
351    *   The result's index or -1 if no result is highlighted.
352    */
353   var getHighlighted = function() {
354     return $('.result', $results).index($('.result.highlight', $results));
355   };
356 
357   /**
358    * Select the current highlighted element, if any.
359    */
360   var select = function() {
361     var $result = $('.result', $results).eq(getHighlighted());
362     if (!$result.length) {
363       return; // No selectable element
364     }
365     var result = $result.data('result');
366     callbacks.select(result, $input);
367     // Redraw again, if the callback changed focus or content
368     reprocess();
369   };
370 
371   /**
372    * Fetch results asynchronously via AJAX.
373    * Errors are ignored.
374    *
375    * @param {String} query
376    *   The query string.
377    */
378   var fetchResults = function(query) {
379     // Synchronously fetch local data
380     if (isLocal) {
381       cacheResults(query, callbacks.queryLocalResults(query, resource,
382                                                       options.caseSensitive));
383       redraw();
384     }
385     // Asynchronously fetch remote data
386     else {
387       activeRemoteCalls.push(query);
388       var url = callbacks.constructURL(resource, query);
389       callbacks.beginFetching($input);
390       callbacks.fetchRemoteData(url, function(data) {
391         var searchResults = callbacks.processRemoteData(data);
392         if (!$.isArray(searchResults)) {
393           searchResults = [];
394         }
395         cacheResults(query, searchResults);
396         // Remove the query from active remote calls, since it's finished
397         activeRemoteCalls = $.grep(activeRemoteCalls, function(value) {
398           return value != query;
399         });
400         if (!activeRemoteCalls.length) {
401           callbacks.finishFetching($input);
402         }
403         redraw();
404       }, options.remoteTimeout, options.crossOrigin);
405     }
406   };
407 
408   /**
409    * Reprocess the contents of the input field, fetch data and redraw if
410    * necessary.
411    */
412   function reprocess() {
413     var query = callbacks.canonicalQuery($input.val(), options.caseSensitive);
414     clearTimeout(timer);
415     // Indicate that timer is inactive
416     timer = null;
417     redraw();
418     if (query.length >= options.charLimit && !$.isArray(cache[query]) &&
419         $.inArray(query, activeRemoteCalls) == -1) {
420       // Fetching is required
421       $results.empty();
422       if (isLocal) {
423         fetchResults(query);
424       }
425       else {
426         timer = setTimeout(function() {
427           fetchResults(query);
428           timer = null;
429         }, options.delay);
430       }
431     }
432   };
433 
434   /**
435    * Redraws the autocomplete list based on current query and focus.
436    *
437    * @param {Boolean} [focus]
438    *   (default=false) Force to treat the input element like it's focused.
439    */
440   var redraw = function(focus) {
441     var query = callbacks.canonicalQuery($input.val(), options.caseSensitive);
442 
443     // The query does not exist in db
444     if (!$.isArray(cache[query])) {
445       lastRenderedQuery = null;
446       $results.empty();
447     }
448     // The query exists and is not already rendered
449     else if (lastRenderedQuery !== query) {
450       lastRenderedQuery = query;
451       renderResults(cache[query]);
452       setHighlighted(0);
453     }
454     // Finally show/hide based on focus and emptiness
455     if (($input.is(':focus') || focus) && !$results.is(':empty')) {
456       $results.filter(':hidden').show() // Show if hidden
457         .scrollTop($results.data('scroll-top')); // Reset the lost scrolling
458     }
459     else if ($results.is(':visible')) {
460       // Store the scrolling position for later
461       $results.data('scroll-top', $results.scrollTop())
462         .hide(); // Hiding it resets it's scrollTop
463     }
464   };
465 
466   /**
467    * Regenerate the DOM content within the results list for a given set of
468    * results. Heavy method, use only when necessary.
469    *
470    * @param {Array[Object]} results
471    *   An array of result objects to render.
472    */
473   var renderResults = function(results) {
474     $results.empty();
475     var groups = {}; // Key is the group name, value is the heading element.
476 
477     $.each(results, function(index, result) {
478       if ($.type(result) != 'object') {
479         return; // Continue
480       }
481 
482       var output = callbacks.themeResult(result);
483       if ($.type(output) != 'string') {
484         return; // Continue
485       }
486 
487       // Add the group if it doesn't exist
488       var group = callbacks.getGroup(result);
489       if ($.type(group) == 'string' && !groups[group]) {
490         var $groupHeading = $('<li />').addClass('group')
491           .append($('<h3 />').html(group))
492           .appendTo($results);
493         groups[group] = $groupHeading;
494       }
495 
496       var $result = $('<li />').addClass('result')
497         .append(output)
498         .data('result', result) // Store the result object on this DOM element
499         .addClass(result.addClass);
500 
501       // First groupless item
502       if ($.type(group) != 'string' &&
503           !$results.children().first().is('.result')) {
504         $results.prepend($result);
505         return; // Continue
506       }
507       var $traverseFrom = ($.type(group) == 'string') ?
508                           groups[group] : $results.children().first();
509       var $target = $traverseFrom.nextUntil('.group').last();
510       $result.insertAfter($target.length ? $target : $traverseFrom);
511     });
512   };
513 };

515 /*
516  * CALLBACK METHODS

519 /**
520  * These callbacks are supposed to be overridden by you when you need
521  * customization of the default behavior. When you are overriding a callback
522  * function, it is a good idea to copy the source code from the default
523  * callback function, as a skeleton.
524  *
525  * @name callbacks
526  * @namespace
527  */
528 var defaultCallbacks = {
529   /**
530    * @lends callbacks.prototype
531    */
532 
533   /**
534    * Gets fired when the user selects a result by clicking or using the
535    * keyboard to select an element.
536    *
537    * <br /><br /><em>Default behavior: Inserts the result's title into the
538    * input field.</em>
539    *
540    * @param {Object} result
541    *   The result object that was selected.
542    *
543    * @param {Object} $input
544    *   The input DOM element, wrapped in jQuery.
545    */
546   select: function(result, $input) {
547     $input.val(result.title);
548   },
549 
550   /**
551    * Given a result object, theme it to HTML.
552    *
553    * <br /><br /><em>Default behavior: Wraps result.title in an h4 tag, and
554    * result.description in a p tag. Note that no sanitization of malicious
555    * scripts is done here. Whatever is within the title/description is just
556    * printed out. May contain HTML.</em>
557    *
558    * @param {Object} result
559    *   The result object that should be rendered.
560    *
561    * @returns {String}
562    *   HTML output, will be wrapped in a list element.
563    */
564   themeResult: function(result) {
565     var output = [];
566     if ($.type(result.title) == 'string') {
567       output.push('<h4>', result.title, '</h4>');
568     }
569     if ($.type(result.description) == 'string') {
570       output.push('<p>', result.description, '</p>');
571     }
572     return output.join('');
573   },
574 
575   /**
576    * Retrieve local results from the local resource by providing a query
577    * string.
578    *
579    * <br /><br /><em>Default behavior: Automatically handles arrays, if the
580    * data inside each element is either a plain string or a result object.
581    * If it is a result object, it will match the query string against the
582    * title and description property. Search is not case sensitive.</em>
583    *
584    * @param {String} query
585    *   The query string, unescaped. May contain any UTF-8 character.
586    *   If case insensitive, it already is lowercased.
587    *
588    * @param {Object} resource
589    *   The resource provided in the {@link jQuery.betterAutocomplete} init
590    *   constructor.
591    *
592    * @param {Boolean} caseSensitive
593    *   From options.caseSensitive, the searching should be case sensitive.
594    *
595    * @returns {Array[Object]}
596    *   A flat array containing pure result objects. Must return an array.
597    */
598   queryLocalResults: function(query, resource, caseSensitive) {
599     if (!$.isArray(resource)) {
600       // Per default Better Autocomplete only handles arrays
601       return [];
602     }
603     var results = [];
604     $.each(resource, function(i, value) {
605       switch ($.type(value)) {
606       case 'string': // Flat array of strings
607         if ((caseSensitive ? value : value.toLowerCase())
608             .indexOf(query) >= 0) {
609           // Match found
610           results.push({ title: value });
611         }
612         break;
613       case 'object': // Array of result objects
614         if ($.type(value.title) == 'string' &&
615             (caseSensitive ? value.title : value.title.toLowerCase())
616             .indexOf(query) >= 0) {
617           // Match found in title field
618           results.push(value);
619         }
620         else if ($.type(value.description) == 'string' &&
621                  (caseSensitive ? value.description :
622                  value.description.toLowerCase()).indexOf(query) >= 0) {
623           // Match found in description field
624           results.push(value);
625         }
626         break;
627       }
628     });
629     return results;
630   },
631 
632   /**
633    * Fetch remote result data and return it using completeCallback when
634    * fetching is finished. Must be asynchronous in order to not freeze the
635    * Better Autocomplete instance.
636    *
637    * <br /><br /><em>Default behavior: Fetches JSON data from the url, using
638    * the jQuery.ajax() method. Errors are ignored.</em>
639    *
640    * @param {String} url
641    *   The URL to fetch data from.
642    *
643    * @param {Function} completeCallback
644    *   This function must be called, even if an error occurs. It takes zero
645    *   or one parameter: the data that was fetched.
646    *
647    * @param {Number} timeout
648    *   The preferred timeout for the request. This callback should respect
649    *   the timeout.
650    *
651    * @param {Boolean} crossOrigin
652    *   True if a cross origin request should be performed.
653    */
654   fetchRemoteData: function(url, completeCallback, timeout, crossOrigin) {
655     $.ajax({
656       url: url,
657       dataType: crossOrigin && !$.support.cors ? 'jsonp' : 'json',
658       timeout: timeout,
659       success: function(data, textStatus) {
660         completeCallback(data);
661       },
662       error: function(jqXHR, textStatus, errorThrown) {
663         completeCallback();
664       }
665     });
666   },
667 
668   /**
669    * Process remote fetched data by extracting an array of result objects
670    * from it. This callback is useful if the fetched data is not the plain
671    * results array, but a more complicated object which does contain results.
672    *
673    * <br /><br /><em>Default behavior: If the data is defined and is an
674    * array, return it. Otherwise return an empty array.</em>
675    *
676    * @param {mixed} data
677    *   The raw data recieved from the server. Can be undefined.
678    *
679    * @returns {Array[Object]}
680    *   A flat array containing result objects. Must return an array.
681    */
682   processRemoteData: function(data) {
683     if ($.isArray(data)) {
684       return data;
685     }
686     else {
687       return [];
688     }
689   },
690 
691   /**
692    * From a given result object, return it's group name (if any). Used for
693    * grouping results together.
694    *
695    * <br /><br /><em>Default behavior: If the result has a "group" property
696    * defined, return it.</em>
697    *
698    * @param {Object} result
699    *   The result object.
700    *
701    * @returns {String}
702    *   The group name, may contain HTML. If no group, don't return anything.
703    */
704   getGroup: function(result) {
705     if ($.type(result.group) == 'string') {
706       return result.group;
707     }
708   },
709 
710   /**
711    * Called when remote fetching begins.
712    *
713    * <br /><br /><em>Default behavior: Adds the CSS class "fetching" to the
714    * input field, for styling purposes.</em>
715    *
716    * @param {Object} $input
717    *   The input DOM element, wrapped in jQuery.
718    */
719   beginFetching: function($input) {
720     $input.addClass('fetching');
721   },
722 
723   /**
724    * Called when fetching is finished. All active requests must finish before
725    * this function is called.
726    *
727    * <br /><br /><em>Default behavior: Removes the "fetching" class.</em>
728    *
729    * @param {Object} $input
730    *   The input DOM element, wrapped in jQuery.
731    */
732   finishFetching: function($input) {
733     $input.removeClass('fetching');
734   },
735 
736   /**
737    * Construct the remote fetching URL.
738    *
739    * <br /><br /><em>Default behavior: Adds "?q=query" to the path. The query
740    * string is URL encoded.</em>
741    *
742    * @param {String} path
